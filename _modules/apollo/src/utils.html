[docs]def read_params_in_from_json(filename):
    """

    :param filename:
    :return:
    """
    root = os.path.dirname(os.getcwd())
    credential_file_path = root + '\\params\\' + filename
    json_string = open(credential_file_path).read()
    params = json.loads(json_string.replace('\n', ' ').replace('\t', ' '))

[docs]def setup_logger(logger_name, log_file, level=logging.INFO):
    """

    :param logger_name:
    :param log_file:
    :param level:
    """
    l = logging.getLogger(logger_name)
    formatter = logging.Formatter('%(asctime)s %(levelname)-8s %(message)s')
    fileHandler = logging.FileHandler(log_file, mode='w')
    fileHandler.setFormatter(formatter)
    streamHandler = logging.StreamHandler()
    streamHandler.setFormatter(formatter)

    l.setLevel(level)
    l.addHandler(fileHandler)
    l.addHandler(streamHandler)

[docs]def wfa(abs_error, actual, forecast, epsilon):
    """

    :param abs_error:
    :param actual:
    :param forecast:
    :param epsilon:
    :return:
    """
    if sum(forecast) == 0 and sum(actual) == 0 and sum(abs_error) == 0:
        wfa = 0
    else:
        wfa = max(0, 1 - 2*sum(abs_error)/(epsilon + sum(forecast) + sum(actual)))
    
    return wfa

[docs]def chunk_dict(d, number_of_chunks):
    """

    :param d:
    :param number_of_chunks:
    :return:
    """
    chunk_size = math.ceil(len(d.items())/number_of_chunks)
    n = 1
    chunk = {}
    list_of_chunks = []
    for key, value in d.items():
        chunk[key] = value
        if (n % chunk_size) == 0:            
            list_of_chunks.append(chunk)
            n = 1
            chunk = {}        
        n += 1
    
    return list_of_chunks
    
    
    
    return chunk

            x = [0]
